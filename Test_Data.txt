import math

def calculate_clock_angle(time):
    # Split the input time into hours and minutes
    hours, minutes = map(int, time.split(':'))

    # Ensure hours are in the range [0, 11] and minutes are in the range [0, 59]
    hours %= 12
    minutes %= 60

    # Calculate the angles made by hour and minute hands with respect to 12:00
    hour_angle = 0.5 * (60 * hours + minutes)
    minute_angle = 6 * minutes

    # Calculate the absolute difference between the angles
    angle = abs(hour_angle - minute_angle)

    # Consider the smaller angle between the hands (360 - angle) if needed
    angle = min(360 - angle, angle)

    return math.ceil(angle)

# Read input from standard input
time = input("Enter the time in HH:MM format: ")

# Calculate and print the smallest angle
result = calculate_clock_angle(time)
print(result)
