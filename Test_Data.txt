using System;

class Program
{
    static void Main()
    {
        // Read input from the user
        Console.Write("Enter the value of N: ");
        int N = int.Parse(Console.ReadLine());

        // Call the function to print the hourglass pattern
        PrintHourglassPattern(N);
    }

    static void PrintHourglassPattern(int N)
    {
        for (int i = 1; i <= N; i++)
        {
            // Print decreasing numbers
            for (int j = i; j >= 1; j--)
            {
                Console.Write(j + " ");
            }

            // Print zeros
            for (int j = 0; j < N - i; j++)
            {
                Console.Write("0 ");
            }

            // Move to the next line
            Console.WriteLine();
        }

        for (int i = N - 1; i >= 1; i--)
        {
            // Print decreasing numbers
            for (int j = i; j >= 1; j--)
            {
                Console.Write(j + " ");
            }

            // Print zeros
            for (int j = 0; j < N - i; j++)
            {
                Console.Write("0 ");
            }

            // Move to the next line
            Console.WriteLine();
        }
    }
}
