using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Read input from standard input
        Console.Write("Enter player IDs separated by comma: ");
        string input = Console.ReadLine();

        // Split the input into an array of player IDs
        string[] playerIDs = input.Split(',');

        // Calculate the winner and print the result
        string winner = CalculateEliminatorXWinner(playerIDs);
        Console.WriteLine($"The winner is: {winner}");
    }

    static string CalculateEliminatorXWinner(string[] playerIDs)
    {
        List<string> remainingPlayers = new List<string>(playerIDs);

        bool eliminateFromLeft = true;

        while (remainingPlayers.Count > 1)
        {
            List<string> nextRound = new List<string>();

            for (int i = 0; i < remainingPlayers.Count; i++)
            {
                if (i % 2 == 0 && eliminateFromLeft)
                {
                    // Eliminate alternate players from left to right
                    nextRound.Add(remainingPlayers[i]);
                }
                else if (i % 2 == 1 && !eliminateFromLeft)
                {
                    // Eliminate alternate players from right to left
                    nextRound.Add(remainingPlayers[i]);
                }
            }

            remainingPlayers = nextRound;
            eliminateFromLeft = !eliminateFromLeft;
        }

        return remainingPlayers[0];
    }
}
