using System;

class Program
{
    static void Main()
    {
        // Read input from standard input
        Console.Write("Enter the time in HH:MM format: ");
        string inputTime = Console.ReadLine();

        // Calculate and print the smallest angle
        int result = CalculateClockAngle(inputTime);
        Console.WriteLine(result);
    }

    static int CalculateClockAngle(string time)
    {
        // Split the input time into hours and minutes
        string[] timeParts = time.Split(':');
        int hours = int.Parse(timeParts[0]);
        int minutes = int.Parse(timeParts[1]);

        // Ensure hours are in the range [0, 11] and minutes are in the range [0, 59]
        hours %= 12;
        minutes %= 60;

        // Calculate the angles made by hour and minute hands with respect to 12:00
        double hourAngle = 0.5 * (60 * hours + minutes);
        double minuteAngle = 6 * minutes;

        // Calculate the absolute difference between the angles
        double angle = Math.Abs(hourAngle - minuteAngle);

        // Consider the smaller angle between the hands (360 - angle) if needed
        angle = Math.Min(360 - angle, angle);

        // Round up to the nearest integer
        return (int)Math.Ceiling(angle);
    }
}
